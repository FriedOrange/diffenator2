# Rules

# Build Hinting docs
rule diffbrowsers
  command = _diffbrowsers diff -fb \$fonts_before -fa \$fonts_after -o \$out \$
      -pt \$pt_size --imgs --filter-styles "\$filters"

# Run diffenator VF
rule diffenator
  command = _diffenator \$font_before \$font_after -t \$threshold -o \$out
rule diffenator-inst
  command = _diffenator \$font_before \$font_after -t \$threshold -o \$out \$
      --coords \$coords

# Build rules
build out/diffbrowsers: diffbrowsers
  fonts_before = \$
      .*/tests/data/MavenPro\[wght\].subset.ttf
  fonts_after = \$
      .*/tests/data/MavenPro\[wght\].subset.mod.ttf
  out = out/diffbrowsers
  pt_size = 20
  filters = Medium|ExtraBold
build out/Bold: diffenator-inst
  font_before = \$
      .*/tests/data/MavenPro\[wght\].subset.ttf
  font_after = \$
      .*/tests/data/MavenPro\[wght\].subset.mod.ttf
  out = Bold
  threshold = 0.9
  coords = wght=700.0
build out/ExtraBold: diffenator-inst
  font_before = \$
      .*/tests/data/MavenPro\[wght\].subset.ttf
  font_after = \$
      .*/tests/data/MavenPro\[wght\].subset.mod.ttf
  out = ExtraBold
  threshold = 0.9
  coords = wght=800.0
build out/Medium: diffenator-inst
  font_before = \$
      .*/tests/data/MavenPro\[wght\].subset.ttf
  font_after = \$
      .*/tests/data/MavenPro\[wght\].subset.mod.ttf
  out = Medium
  threshold = 0.9
  coords = wght=500.0
